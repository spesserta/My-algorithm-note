#include<iostream>
#define MAXSIZE 10
//顺序表结构体定义 
typedef struct{
	int data[MAXSIZE];
	int length;
}SqList;

//初始化/清空顺序表
bool InitSqList(SqList &L)
{
	for(int i=0;i<MAXSIZE;i++)
	   L.data[i]=0;
    L.length=0;
    return true;
} 

//显示顺序表
void ShowSqListt(SqList L)
{
	if(L.length==0)   std::cout<<"当前表中没有元素！"<<std::endl;
	else{
	   std::cout<<"当前顺序表表长为："<<L.length<<std::endl<<"元素有：";
   	   for(int i=0;i<L.length;i++)
	   std::cout<<L.data[i]<<" ";
   }  
	
}

//添加一个表尾元素 
bool addSqList(SqList &L,int adata)
{
	if(L.length==MAXSIZE) return false;
	L.data[L.length]=adata;
	L.length++;
	return true;	
}
 
//删除表尾元素
bool DeSqList(SqList &L)
{
	if(L.length==0)  return false;
	int x=L.data[L.length-1];
	L.data[L.length-1]=0;
	L.length--;
	std::cout<<"删除成功!删除元素值为:"<<x<<std::endl;
	return true;
} 
 
//插入一个元素
bool InSertSqList(SqList &L,int i,int adata)
{
   	if(L.length>=MAXSIZE)  return false;
    if(i<1||i>L.length+1) return false;
    for(int j=L.length-1;j>=i-1;j--)
      L.data[j+1]=L.data[j];
    L.data[i-1]=adata;
    return true;
} 

//删除指定位置元素
bool DeleteSqList(SqList &L,int i)
{
	int x;
	if(i<1||i>L.length)  return false;
	if(L.length==0)  return false;
	x=L.data[i-1];
	for(int j=i;j<L.length-1;j++)
	  L.data[j-1]=L.data[j];
    L.length--;
	std::cout<<"删除成功!删除元素值为:"<<x<<std::endl;
} 

//按位查找
int SearchSqList(SqList L,int i)
{
	return L.data[i-1];
} 

//按值查找
int SeSqList(SqList L,int e)
{
	for(int i=0;i<L.length;i++)
	   if(L.data[i]==e)
	     return i+1;
    return -1;
}
