#include<stdio.h>
#include<stdlib.h>
using namespace std;
typedef struct LNode{   //结构体 
	struct LNode *next;
	char data;
}LNode;

LNode *InitLinkList()    //初始化 
{
	LNode *L=(LNode*)malloc(sizeof(LNode));
	L->next=NULL;
	return L;
}

//最后面增加节点
void addNode(LNode *L)  //传入头结点 
{
    LNode *tail=L;
	while(tail->next!=NULL)
      tail=tail->next;
    
	char temp='a';
	while(temp!='\n')
	{
		temp=getchar();
		LNode *s=(LNode*)malloc(sizeof(LNode));
		s->data=temp;
		s->next=tail->next;
		tail->next=s;
	}
	
}

//返回节点个数
int showLNode(LNode *L)
{
	int sum=0;
	LNode *p=L;
	while(p->next!=NULL)
	{
		p=p->next;
		sum++;
	}
	return sum;
	
}

	int sum=0;

		printf("%c ",p->data);
